\begin{frame}{Motivation}
\small
\textcolor{TealAccent}{Key claim: unequal measurement can create unequal decisions before a model ever looks biased.}

\vspace{0.45em}
\begin{columns}[T,onlytextwidth]
  \column{0.60\textwidth}
    \begin{block}{Two domains, same structure}
      \begin{itemize}
        \item \textbf{Clinical:} choose tests or escalation steps under partial feedback.
        \item \textbf{Quantum routing:} choose paths and allocate scarce resources while network conditions shift.
        \item In both settings, some groups may systematically receive weaker, delayed, or noisier context.
        \item If we optimize only aggregate reward, subgroup failures can remain hidden.
      \end{itemize}
    \end{block}
  \column{0.38\textwidth}
    \begin{block}{Fairness examples}
      \textbf{Clinical:} FNR/FPR gaps, delayed escalation.\\[0.3em]
      \textbf{Routing:} success and latency gaps across user or flow groups.
    \end{block}
    \begin{block}{Why bandits?}
      Bandits model \textbf{partial feedback}: you only observe the outcome of the action you chose.
    \end{block}
\end{columns}

\refs{Fairness targets: Hardt et al. (equality of opportunity); Joseph et al. (fair bandits).}
\end{frame}

\begin{frame}{Background: Bandit Learning and Two Testbeds}
\small
\textcolor{TealAccent}{The project studies how different levels of decision context affect fairness, equity, and performance over time.}

\vspace{0.35em}
\begin{columns}[T,onlytextwidth]
  \column{0.31\textwidth}
    \begin{block}{Non-contextual bandits}
      Use reward history alone.\\[0.2em]
      \textcolor{SoftGray}{Examples: $\epsilon$-greedy, UCB, Thompson sampling.}
    \end{block}
  \column{0.31\textwidth}
    \begin{block}{Contextual bandits}
      Use current features before acting.\\[0.2em]
      \textcolor{SoftGray}{Example: LinUCB-style policies.}
    \end{block}
  \column{0.31\textwidth}
    \begin{block}{Informative context}
      Use richer or selectively acquired context.\\[0.2em]
      \textcolor{SoftGray}{Useful when context quality is itself uneven.}
    \end{block}
\end{columns}

\vspace{0.35em}
\begin{columns}[T,onlytextwidth]
  \column{0.49\textwidth}
    \begin{block}{Clinical diagnostic workflows}
      Decisions: test selection, follow-up escalation, or model/pipeline choice.\\[0.2em]
      \textcolor{SoftGray}{Fairness concern: subgroup error gaps and delayed care.}
    \end{block}
  \column{0.49\textwidth}
    \begin{block}{Quantum routing and resource allocation}
      Decisions: path selection and scarce-resource allocation under uncertain link success.\\[0.2em]
      \textcolor{SoftGray}{Fairness concern: service-equity gaps in success, latency, or access.}
    \end{block}
\end{columns}

\refs{Bandit foundations: Auer et al., Li et al.; domain framing from the revised proposal and survey.}
\end{frame}

\begin{frame}{Related Work: Bandits, Limited Context, and Shift}
\small
\textcolor{TealAccent}{These works explain how policies learn online, and why limited context plus shift is the right stress test here.}

\vspace{0.35em}
\begin{columns}[T,onlytextwidth]
  \column{0.32\textwidth}
    \begin{block}{What the work does}
      UCB, Thompson sampling, and LinUCB establish the main non-contextual and contextual learning baselines.
    \end{block}
  \column{0.32\textwidth}
    \begin{block}{Why it matters here}
      Restricted-context, unobserved-context, and non-stationary bandits motivate uneven information, missingness, and drift in both domains.
    \end{block}
  \column{0.32\textwidth}
    \begin{block}{What remains open}
      Most of this literature still optimizes utility, not time-evolving disparity or worst-group outcomes.
    \end{block}
\end{columns}

\refs{Auer et al.; Li et al.; Besbes et al.; Garivier \& Moulines; restricted-context bandit literature.}
\end{frame}

\begin{frame}{Related Work: Fairness in Sequential Learning}
\small
\textcolor{TealAccent}{Fairness work gives the measurement targets and mitigation logic, but not yet the full cross-domain framework.}

\vspace{0.35em}
\begin{columns}[T,onlytextwidth]
  \column{0.32\textwidth}
    \begin{block}{What the work does}
      Equality of opportunity, clinical bias work, fair bandits, and fair RL define fairness targets and constraints for learning systems.
    \end{block}
  \column{0.32\textwidth}
    \begin{block}{Why it matters here}
      These papers justify tracking FNR/FPR gaps, delayed escalation gaps, and service-equity gaps during learning rather than only after deployment.
    \end{block}
  \column{0.32\textwidth}
    \begin{block}{What remains open}
      Fewer works study outcome-based disparity curves under distribution shift and differing context quality across groups.
    \end{block}
\end{columns}

\refs{Hardt et al.; Joseph et al.; Chen et al.; broader fair RL and clinical-bias literature.}
\end{frame}

\begin{frame}{Related Work: Domain Testbeds}
\small
\textcolor{TealAccent}{The project is grounded in one fairness-oriented diagnostic line of work and one quantum routing line of work.}

\vspace{0.35em}
\begin{columns}[T,onlytextwidth]
  \column{0.49\textwidth}
    \begin{block}{EQUITAS / equitable bioinformatics}
      \begin{itemize}
        \item Gives the project a fairness-audit and mitigation perspective in diagnostic-relevant settings.
        \item Shows why predictive performance and fairness should be evaluated together.
        \item Limitation: primarily offline/static rather than online sequential decision-making.
      \end{itemize}
    \end{block}
  \column{0.49\textwidth}
    \begin{block}{Threat-aware quantum routing}
      \begin{itemize}
        \item Provides a realistic routing testbed with uncertainty, resource scarcity, and adversarial conditions.
        \item Shows bandit-style methods can work in quantum networking.
        \item Limitation: robustness is central, but explicit service-equity tracking is not yet the focus.
      \end{itemize}
    \end{block}
\end{columns}

\refs{EQUITAS paper; threat-aware quantum routing paper; network context from Wehner, Pant, and Caleffi.}

\begin{frame}{Proposed Project: Core Research Question}
\small
\textcolor{TealAccent}{Goal: learn a policy that is high utility and low disparity under missing context and shift.}

\vspace{0.3em}
\begin{block}{Compressed formal problem}
At round $t$, observe group $g_t \in \mathcal{G}$ and degraded context $\tilde{x}_t$, choose action $a_t \in \{1,\dots,K\}$, and observe only the reward of the chosen action.
\[
\tilde{x}_t = M_t(g_t)\odot x_t
\]
where $M_t(g_t)$ models group-dependent missingness or noise.
\end{block}

\vspace{0.25em}
\begin{columns}[T,onlytextwidth]
  \column{0.49\textwidth}
    \begin{block}{Utility objective}
      Maximize expected reward or minimize regret under partial feedback.
    \end{block}
  \column{0.49\textwidth}
    \begin{block}{Fairness monitoring}
      Track time-evolving disparity windows, e.g. equality-of-opportunity gaps or service-equity gaps.
    \end{block}
\end{columns}

\refs{Li et al.; Hardt et al.; Besbes et al.; Garivier \& Moulines.}
\end{frame}

\begin{frame}{Proposed Project: Two Testbeds to Build}
\scriptsize
\textcolor{TealAccent}{One simulation-first clinical environment plus one quantum routing environment with the same evaluation logic.}

\vspace{0.2em}
\begin{columns}[T,onlytextwidth]
  \column{0.54\textwidth}
    \begin{block}{Clinical testbed}
      \begin{itemize}
        \item State: patient context plus a missingness mask.
        \item Actions: next test, model, or escalation step.
        \item Metrics: utility, delay cost, and group disparity curves.
      \end{itemize}
    \end{block}
    \begin{block}{Quantum testbed}
      \begin{itemize}
        \item Paths or path-groups act like arms under uncertain link success.
        \item Regimes include stochastic and adaptive threat conditions.
        \item Metrics: success, latency, and access gaps across groups.
      \end{itemize}
    \end{block}

\begin{frame}{Takeaway}
\small
\textcolor{TealAccent}{The goal is not just a model, but a reusable fairness-aware decision framework.}

\vspace{0.3em}
\begin{block}{Main message}
Context is not just ``more features.'' It is a resource whose unequal quality can create persistent outcome gaps unless fairness is monitored and mitigated during learning.
\end{block}

\vspace{0.25em}
\begin{itemize}
  \item Shared framework across two structurally similar but technically different domains.
  \item Evidence for when better context reduces disparity and when explicit mitigation is still required.
  \item A practical bridge from course project to dissertation-scale research.
\end{itemize}
